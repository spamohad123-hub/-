<script>

let N = 7;


function generateInputs() {

  N = parseInt(document.getElementById("numPeople").value);

  const container = document.getElementById("peopleContainer");

  container.innerHTML = "";


  for (let i = 1; i <= N; i++) {

    const div = document.createElement("div");

    div.className = "person-box";


    div.innerHTML = `

      <h3>Person ${i}${i <= 3 ? " (Anchor)" : ""}</h3>

      <input placeholder="Pick 1" id="p${i}a" />

      <input placeholder="Pick 2" id="p${i}b" />

      <input placeholder="Pick 3" id="p${i}c" />

    `;


    container.appendChild(div);

  }

}


// NEW: Dynamically generates the Anchor Strategy for any N >= 4

function fillRecommended() {

  for (let i = 1; i <= N; i++) {

    let picks = [];

    

    // The three anchors cross-reference each other, and use Person 4 as their 3rd pick

    if (i === 1) picks = [2, 3, 4];

    else if (i === 2) picks = [1, 3, 4];

    else if (i === 3) picks = [1, 2, 4];

    // Everyone else points all their choices at the three Anchors

    else picks = [1, 2, 3];


    document.getElementById(`p${i}a`).value = picks[0];

    document.getElementById(`p${i}b`).value = picks[1];

    document.getElementById(`p${i}c`).value = picks[2];

  }

}


function getChoices() {

  const choices = {};

  for (let i = 1; i <= N; i++) {

    const a = parseInt(document.getElementById(`p${i}a`).value);

    const b = parseInt(document.getElementById(`p${i}b`).value);

    const c = parseInt(document.getElementById(`p${i}c`).value);

    choices[i] = [a, b, c].filter(x => !isNaN(x));

  }

  return choices;

}


function generatePartitions(arr) {

  if (arr.length === 0) return [[]];

  const first = arr[0];

  const rest = generatePartitions(arr.slice(1));

  const result = [];

  rest.forEach(partition => {

    for (let i = 0; i < partition.length; i++) {

      const newPartition = partition.map(g => [...g]);

      newPartition[i].unshift(first);

      result.push(newPartition);

    }

    result.push([[first], ...partition]);

  });

  return result;

}


function isValidPartition(partition, choices) {

  for (const group of partition) {

    for (const person of group) {

      const picks = choices[person] || [];

      const valid = picks.some(friend => group.includes(friend));

      if (!valid) return false;

    }

  }

  return true;

}


function normalizePartition(partition) {

  const sortedGroups = partition.map(g => [...g].sort((a,b)=>a-b));

  sortedGroups.sort((a,b)=>a[0]-b[0]);

  return sortedGroups;

}


function runSimulation() {

  const choices = getChoices();

  const people = Array.from({length: N}, (_, i) => i + 1);


  let partitions = generatePartitions(people);

  let validPartitions = [];

  let seen = new Set();


  partitions.forEach(p => {

    if (isValidPartition(p, choices)) {

      const norm = normalizePartition(p);

      const key = JSON.stringify(norm);

      if (!seen.has(key)) {

        seen.add(key);

        validPartitions.push(norm);

      }

    }

  });


  const patternCounts = {};

  validPartitions.forEach(p => {

    const sizes = p.map(g => g.length).sort((a,b)=>b-a);

    const key = JSON.stringify(sizes);

    patternCounts[key] = (patternCounts[key] || 0) + 1;

  });


  let html = `<h3>Total valid groupings: ${validPartitions.length}</h3>`;


  html += `<h3>All Valid Partitions</h3><pre>`;

  validPartitions.forEach(p => {

    html += p.map(g => `(${g.join(", ")})`).join(" ") + "\n";

  });

  html += `</pre>`;


  html += `<h3>Summary</h3><ul>`;

  Object.keys(patternCounts).forEach(key => {

    const count = patternCounts[key];

    const percent = ((count / validPartitions.length) * 100).toFixed(2);

    html += `<li>${key}: ${count} (${percent}%)</li>`;

  });

  html += `</ul>`;


  document.getElementById("output").innerHTML = html;

}


// Initialize with default inputs

generateInputs();
